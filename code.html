            //Get the values from the page 
            //Start or controller function
            function getValues() {
            //Get values from the page 
            let startValue= document.getElementById("startValue").value;
            let endValue= document.getElementById("endValue").value;



            //We need to validate our input
            //parse into integers
            startValue = parseInt(startValue);
            endValue = parseInt(endValue);

            if(Number.isInteger(startValue) && Number.isInteger(endValue)) {

                //We Call generateNumbers 
            let numbers = generateNumbers(startValue, endValue);
                //We Call displayNumbers
                displayNumbers(numbers);

            } else {
                alert("You must enter integers")
            }






            }




            //Generate numbers from the start value to the end value
            //Logic function 

            function generateNumbers(sValue, eValue) {

                let numbers = [];

                //We want to get all numbers from start to end
                // index = index +1 is the same thing as index++
                for(let index = sValue; index <= eValue; index++){

                    //this will excute into a loop until index equals eValue

                        numbers.push(index);
                }

                return numbers;

            }

            //Display the numbers and mark the even numbers in bold
            //Display or view function

            function displayNumbers(numbers){

                let templateRows = "";
                for (let index = 0; index < numbers.length; index++) {

                    let className = "even"

                    let number = numbers[index];

                    if(number % 2 == 0) {

                        className="even"

                    } else{
                        className="odd"
                    }
                    //This does render correctly with Prisim. See the source.
                    templateRows += `${number}`;
                    
                }

                document.getElementById("results").innerHTML = templateRows;
            }
The Code

The getValue function in the brackets has two variables that get the start and end numbers that are submitted from the UI.

The startValue and endValue variable use the parsnt to convert the content of th e first variables into number format instead of a string.

The IF STATEMENT is a condition statement that checks if both variables are numbers. If they are not both then the condition will lead to the else statement.

The else statement in this conditional statement pushes the alert box to the page telling the operator to put in a interger

The generateNumbers Function takes in the two arguments from the startValue and endValue into the parameters and uses it in the index inside the for loop to generate numbers that are within the range typed in the UI.

The displayNumbers function holds a parameter of "numbers" that is an array which inside the function itself is a FOR LOOP that adds all the amount of items in the array and pushes it to the template inside the app.html. As the numbers are listed the last part of the fucntion is a IF statement that determines whether the numbers are bold or not by using a class selector that marks any value if even bold and any odd non bold.
